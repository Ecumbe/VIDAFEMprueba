    get_file_base64: true,

CAMBIO 3 - Matriz de rol permitido por accion
Archivo: backend/API.gs.txt

function isSessionRoleAllowedForAction_(action, role) {
  const act = String(action || "").trim();
  const r = String(role || "").toLowerCase();
  if (!act || !r) return false;

  if (act.indexOf("superadmin_") === 0) return r === "superadmin";
  if (act === "self_update_admin_profile") return r === "admin";
  if (act === "self_update_patient_profile") return r === "paciente";
  if (act === "save_promotion" || act === "get_promo_list" || act === "delete_promotion") {
    return r === "admin" || r === "superadmin";
  }
  if (act === "self_update_password" || act === "first_login_keep" || act === "first_login_update_password") {
    return r === "admin" || r === "paciente";
  }
  if (act === "get_file_base64") return r === "admin" || r === "paciente" || r === "superadmin";
  return true;
}

CAMBIO 4 - Respaldo en primer login para enviar session_token
Archivo: js/auth.js

async function firstLoginApiCall_(payload) {
  const body = Object.assign({}, payload || {});
  if (body.action !== "login" && !body.session_token) {
    try {
      const raw = sessionStorage.getItem("vidafem_session");
      const s = raw ? JSON.parse(raw) : null;
      if (s && s.session_token) body.session_token = s.session_token;
    } catch (e) {}
  }
  const r = await fetch(API_URL, {
    method: "POST",
    body: JSON.stringify(body)
  });
  return r.json();
}

Impacto esperado
- Endpoints sensibles ya no aceptan llamada sin sesion valida.
- Un token de paciente no puede ejecutar acciones de admin/superadmin.
- Se mantiene compatibilidad del resto del sistema (enforcement no se aplico a todo el API).

Nota operativa
- Si habia usuarios con sesion antigua sin token y van a una accion sensible, el sistema pedira re-login.
